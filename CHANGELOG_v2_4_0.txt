  ADD COLUMN IF NOT EXISTS contact_firstname text,
  ADD COLUMN IF NOT EXISTS contact_lastname text,

CREATE TABLE IF NOT EXISTS public.phone_book (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name text, org text,
  phones jsonb DEFAULT '[]',
  emails jsonb DEFAULT '[]',
  category text DEFAULT 'Makler',
  created_by uuid REFERENCES public.staff(id),
  created_at timestamptz DEFAULT now()
);
ALTER TABLE public.phone_book ENABLE ROW LEVEL SECURITY;
CREATE POLICY "all_access" ON public.phone_book FOR ALL USING (true) WITH CHECK (true);
